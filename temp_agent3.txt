  function lerpAngle(prev, next, alpha){
    const TAU = Math.PI * 2;
    let d = (next - prev) % TAU;
    if (d > Math.PI) d -= TAU;
    if (d < -Math.PI) d += TAU;
    return prev + d * Math.max(0, Math.min(1, alpha));
